9std::string trim(const std::string& s) {

12 auto end = s.find_last_not_of(" \t\r\n");

26// Split a CSV line; double-quoted fields may contain commas.
27std::vector<std::string> splitCSV(const std::string& line) {
28 std::vector<std::string> fields;
29 std::string field;
30 bool inQuotes = false;
31 for (char c : line) {
32 if (c == '"') inQuotes = !inQuotes;
33 else if (c == ',' && !inQuotes) { fields.push_back(trim(field)); field.clear(); }
34 else field += c;
35 }
36 fields.push_back(trim(field));
37 return fields;
38}
