194/*
195 * Auxiliary function to remove an outgoing edge (with a given destination (d))
196 * from a vertex (this).
197 * Returns true if successful, and false if such edge does not exist.
198 */
199template <class T>
200bool Vertex<T>::removeEdge(T in)
201{
202 bool removedEdge = false;
203 auto it = adj.begin();
204 while (it != adj.end())
205 {
206 Edge<T> *edge = *it;
207 Vertex<T> *dest = edge->getDest();
208 if (dest->getInfo() == in)
209 {
210 it = adj.erase(it);
211 deleteEdge(edge);
212 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
213 }
214 else
215 {
216 it++;
217 }
218 }
219 return removedEdge;
220}

222/*
223 * Auxiliary function to remove an outgoing edge of a vertex.
224 */
225template <class T>
226void Vertex<T>::removeOutgoingEdges()
227{
228 auto it = adj.begin();
229 while (it != adj.end())
230 {
231 Edge<T> *edge = *it;
232 it = adj.erase(it);
233 deleteEdge(edge);
234 }
235}

345template <class T>
346void Vertex<T>::deleteEdge(Edge<T> *edge)
347{
348 Vertex<T> *dest = edge->getDest();
349 // Remove the corresponding edge from the incoming list
350 auto it = dest->incoming.begin();
351 while (it != dest->incoming.end())
352 {
353 if ((*it)->getOrig()->getInfo() == info)
354 {
355 it = dest->incoming.erase(it);
356 }
357 else
358 {
359 it++;
360 }
361 }
362 delete edge;
363}

474/*
475 * Removes a vertex with a given content (in) from a graph (this), and
476 * all outgoing and incoming edges.
477 * Returns true if successful, and false if such vertex does not exist.
478 */
479template <class T>
480bool Graph<T>::removeVertex(const T &in)
481{
482 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++)
483 {
484 if ((*it)->getInfo() == in)
485 {
486 auto v = *it;
487 v->removeOutgoingEdges();
488 for (auto u : vertexSet)
489 {
490 u->removeEdge(v->getInfo());
491 }
492 vertexSet.erase(it);
493 delete v;
494 return true;
495 }
496 }
497 return false;
498}

500/*
501 * Adds an edge to a graph (this), given the contents of the source and
502 * destination vertices and the edge weight (w).
503 * Returns true if successful, and false if the source or destination vertex does not exist.
504 */
505template <class T>
506bool Graph<T>::addEdge(const T &sourc, const T &dest, double w)
507{
508 auto v1 = findVertex(sourc);
509 auto v2 = findVertex(dest);
510 if (v1 == nullptr || v2 == nullptr)
511 return false;
512 v1->addEdge(v2, w);
513 return true;
514}

516/*
517 * Removes an edge from a graph (this).
518 * The edge is identified by the source (sourc) and destination (dest) contents.
519 * Returns true if successful, and false if such edge does not exist.
520 */
521template <class T>
522bool Graph<T>::removeEdge(const T &sourc, const T &dest)
523{
524 Vertex<T> *srcVertex = findVertex(sourc);
525 if (srcVertex == nullptr)
526 {
527 return false;
528 }
529 return srcVertex->removeEdge(dest);

532template <class T>
533bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double w)
534{
535 auto v1 = findVertex(sourc);
536 auto v2 = findVertex(dest);
537 if (v1 == nullptr || v2 == nullptr)
538 return false;
539 auto e1 = v1->addEdge(v2, w);
540 auto e2 = v2->addEdge(v1, w);
541 e1->setReverse(e2);
542 e2->setReverse(e1);
543 return true;
544}

577 if (this != &other) {

580 for (auto v : vertexSet) delete v;
