34int createGraphs::getMatchedDomain(const Submission& sub, const Reviewer& rev, int mode) {
35 // Mode 0 behaves like mode 1 for domain matching (output is just suppressed).
36 int m = (mode == 0) ? 1 : mode;
37
38 // Build applicable topic lists for sub and rev according to mode.
39 vector<int> subTopics, revTopics;
40
41 subTopics.push_back(sub.primaryTopic);
42 if (m >= 2 && sub.secondaryTopic != -1)
43 subTopics.push_back(sub.secondaryTopic);
44
45 revTopics.push_back(rev.primaryTopic);
46 if (m == 3 && rev.secondaryTopic != -1)
47 revTopics.push_back(rev.secondaryTopic);
48
49 // Return first matching topic (primary topics tried first).
50 for (int st : subTopics)
51 for (int rt : revTopics)
52 if (st == rt) return st;
53
54 return -1; // no match
55}

57Graph<int> createGraphs::buildReviewFlowGraph(
58 const vector<Submission>& submissions,
59 const vector<Reviewer>& reviewers,
60 const Parameters& params,
61 int mode) {
62
63 int P = (int)submissions.size();
64 int R = (int)reviewers.size();
65 int src = sourceId();
66 int snk = sinkId(P, R);
67
68 Graph<int> g;
69
70 // --- Add vertices ---
71 g.addVertex(src);
72 for (int i = 0; i < P; ++i)
73 g.addVertex(submissionNodeId(i));
74 for (int j = 0; j < R; ++j)
75 g.addVertex(reviewerNodeId(j, P));
76 g.addVertex(snk);
77
78 // --- Edges: source -> submission ---
79 // Capacity = MinReviewsPerSubmission: bounds the maximum flow (and thus
80 // reviews) that can reach each submission. A complete assignment is
81 // achieved when total flow == P * MinReviewsPerSubmission.
82 for (int i = 0; i < P; ++i)
83 g.addEdge(src, submissionNodeId(i), params.MinReviewsPerSubmission);
84
85 // --- Edges: submission -> reviewer (capacity 1) ---
86 // An edge exists when the reviewer can review the submission according to
87 // the current mode:
88 // Mode 0/1: sub.primary == rev.primary
89 // Mode 2 : (sub.primary OR sub.secondary) == rev.primary
90 // Mode 3 : (sub.primary OR sub.secondary) == (rev.primary OR rev.secondary)
91 for (int i = 0; i < P; ++i) {
92 int subNode = submissionNodeId(i);
93 for (int j = 0; j < R; ++j) {
94 if (getMatchedDomain(submissions[i], reviewers[j], mode) != -1)
95 g.addEdge(subNode, reviewerNodeId(j, P), 1);
96 }
97 }
98
99 // --- Edges: reviewer -> sink ---
100 // Each reviewer can review at most MaxReviewsPerReviewer submissions.
101 for (int j = 0; j < R; ++j)
102 g.addEdge(reviewerNodeId(j, P), snk, params.MaxReviewsPerReviewer);
103
104 return g;
105}

111string createGraphs::getNodeLabel(int id, const vector<Submission>& submissions, const vector<Reviewer>& reviewers) {
112 int P = (int)submissions.size();
113 int R = (int)reviewers.size();
114 int src = sourceId();
115 int snk = sinkId(P, R);
116
117 if (id == src) return "SOURCE";

119 for (int i = 0; i < P; ++i)
120 if (id == submissionNodeId(i))
121 return "SUB#" + to_string(submissions[i].id);
122 for (int j = 0; j < R; ++j)
123 if (id == reviewerNodeId(j, P))
124 return "REV#" + to_string(reviewers[j].id);
125 return "NODE#" + to_string(id);

128void createGraphs::printFlowGraph(
129 const Graph<int>& g,
130 const vector<Submission>& submissions,
131 const vector<Reviewer>& reviewers) {
132
133 int P = (int)submissions.size();
134 int R = (int)reviewers.size();
135 int src = sourceId();
136 int snk = sinkId(P, R);
137
138 // Build a label map: node ID -> display name
139 auto label = [&](int id) -> string {
140 return getNodeLabel(id, submissions, reviewers);
141 };
142
143 // Determine column width for alignment
144 size_t maxLabelLen = 6; // "SOURCE" / "SINK"
145 for (const auto& s : submissions)
146 maxLabelLen = max(maxLabelLen, string("SUB#").size() + to_string(s.id).size());
147 for (const auto& r : reviewers)
148 maxLabelLen = max(maxLabelLen, string("REV#").size() + to_string(r.id).size());
149
150 auto pad = [&](const string& s) -> string {
151 return s + string(maxLabelLen - s.size() + 1, ' ');
152 };
153
154 cout << string(60, '-') << "\n";
155 cout << " Flow Graph (" << g.getNumVertex() << " nodes)\n";
156 cout << string(60, '-') << "\n";
157 cout << " " << pad("FROM") << "--> " << pad("TO") << " cap flow\n";
158 cout << string(60, '-') << "\n";
159
160 // Iterate vertices in a predictable order: src, subs, revs, sink
161 vector<int> order;
162 order.push_back(src);
163 for (int i = 0; i < P; ++i) order.push_back(submissionNodeId(i));
164 for (int j = 0; j < R; ++j) order.push_back(reviewerNodeId(j, P));
165 order.push_back(snk);
166
167 for (int nodeId : order) {
168 Vertex<int>* v = g.findVertex(nodeId);
169 if (v == nullptr) continue;
170 for (Edge<int>* e : v->getAdj()) {
171 int destId = e->getDest()->getInfo();
172 double cap = e->getWeight();
173 double flow = e->getFlow();
174 cout << " " << pad(label(nodeId))
175 << "--> " << pad(label(destId))
176 << " " << (int)cap
177 << " " << (int)flow << "\n";
178 }
179 }
180 cout << string(60, '-') << "\n";
181}
