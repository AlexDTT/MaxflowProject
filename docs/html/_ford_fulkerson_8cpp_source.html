5// Recursive DFS helper to find an augmenting path from v to t
6template <class T>
7bool dfsAugment(Vertex<T> *v, Vertex<T> *t)
8{
9 v->setVisited(true);
10
11 if (v == t)
12 return true;
13
14 for (auto e : v->getAdj())
15 {
16 Vertex<T> *w = e->getDest();
17 double residual = e->getWeight() - e->getFlow();
18
19 if (!w->isVisited() && residual > 0)

21 w->setPath(e);
22 if (dfsAugment(w, t))

27 return false;
28}

42// Function to find the minimum residual capacity along the augmenting path
43template <class T>
44double ffFindMinResidualAlongPath(Vertex<T> *s, Vertex<T> *t)

77// Main function implementing the Ford-Fulkerson algorithm
78template <class T>
79void fordFulkerson(Graph<T> *g, T source, T target, std::function<void(const std::vector<T> &, double)> onAugment)
80{
81 Vertex<T> *s = g->findVertex(source);
82 Vertex<T> *t = g->findVertex(target);
83
84 if (s == nullptr || t == nullptr || s == t)
85 throw std::logic_error("Invalid source and/or target vertex");
86
87 for (auto v : g->getVertexSet())
88 for (auto e : v->getAdj())
89 e->setFlow(0);
90 std::vector<Edge<T> *> originalEdges;
91 for (auto v : g->getVertexSet())
92 for (auto e : v->getAdj())
93 originalEdges.push_back(e);
94
95 for (auto e : originalEdges)
96 {
97 if (e->getReverse() == nullptr)
98 {
99 Edge<T> *rev = e->getDest()->addEdge(e->getOrig(), 0);
100 rev->setFlow(0);
101 e->setReverse(rev);
102 rev->setReverse(e);
103 }
104 }
105
106 while (ffFindAugmentingPath(g, s, t))
107 {
108 double f = ffFindMinResidualAlongPath(s, t);
109
110 if (onAugment)
111 {
112 std::vector<T> pathNodes;
113 Vertex<T> *v = t;
114 while (v != s)
115 {
116 pathNodes.push_back(v->getInfo());
117 v = v->getPath()->getOrig();
118 }
119 pathNodes.push_back(s->getInfo());
120 std::reverse(pathNodes.begin(), pathNodes.end());
121 onAugment(pathNodes, f);
122 }
123
124 ffAugmentFlowAlongPath(s, t, f);
125 }
126
127 for (auto v : g->getVertexSet())
128 {
129 std::vector<T> toRemove;
130 for (auto e : v->getAdj())
131 {
132 if (e->getWeight() == 0)
133 {
134 toRemove.push_back(e->getDest()->getInfo());
135 }
136 }
137 for (auto &dest : toRemove)
138 {
139 v->removeEdge(dest);
140 }
141 }
142}
