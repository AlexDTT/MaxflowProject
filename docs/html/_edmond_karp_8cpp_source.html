17// Function to find an augmenting path using Breadth-First Search
18template <class T>
19bool findAugmentingPath(Graph<T> *g, Vertex<T> *s, Vertex<T> *t) {
20 // Mark all vertices as not visited
21 for(auto v : g->getVertexSet()) {
22 v->setVisited(false);
23 }
24
25 std::queue< Vertex<T>*> q;
26 q.push(s);
27 s->setVisited(true);
28

30 Vertex<T> *v = q.front();
31 q.pop();
32

35 Vertex<T> *w = e->getDest();

83template <class T>
84void edmondsKarp(Graph<T> *g, T source, T target, std::function<void(const std::vector<T>&, double)> onAugment) {
85 // Find source and target vertices in the graph
86 Vertex<T>* s = g->findVertex(source);
87 Vertex<T>* t = g->findVertex(target);
88
89 if (s == nullptr || t == nullptr || s == t)
90 throw std::logic_error("Invalid source and/or target vertex");
91
92
93
94 for (auto v : g->getVertexSet())
95 for (auto e : v->getAdj())
96 e->setFlow(0);
97 std::vector<Edge<T>*> originalEdges;
98 for (auto v : g->getVertexSet())
99 for (auto e : v->getAdj())
100 originalEdges.push_back(e);
101
102 for (auto e : originalEdges) {
103 if (e->getReverse() == nullptr) {
104 Edge<T>* rev = e->getDest()->addEdge(e->getOrig(), 0);
105 rev->setFlow(0);
106 e->setReverse(rev);
107 rev->setReverse(e);
108 }
109 }
110
111 while(findAugmentingPath(g, s, t)) {
112 double f = findMinResidualAlongPath(s, t);
113
114 if (onAugment) {
115 std::vector<T> pathNodes;
116 Vertex<T> *v = t;
117 while (v != s) {
118 pathNodes.push_back(v->getInfo());
119 v = v->getPath()->getOrig();
120 }
121 pathNodes.push_back(s->getInfo());
122 std::reverse(pathNodes.begin(), pathNodes.end());
123 onAugment(pathNodes, f);
124 }
125
126 augmentFlowAlongPath(s, t, f);
127 }
128
129
130
131 for (auto v : g->getVertexSet()) {
132 std::vector<T> toRemove;
133 for (auto e : v->getAdj()) {
134 if (e->getWeight() == 0) {
135 toRemove.push_back(e->getDest()->getInfo());
136 }
137 }
138 for (auto& dest : toRemove) {
139 v->removeEdge(dest);
140 }
141 }
